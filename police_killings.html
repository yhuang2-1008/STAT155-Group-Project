library(fivethirtyeight)
library(tidyverse)
library(ggridges)
library(GGally)
library(moderndive)
library(infer)
theme_set(theme_minimal())



  1. Brief overview of the data set police_killings, raceethnicity filled by gender. The graph shows that there is a significantly more males who were killed by police than females. More white males are killed by police, followed by Black males, and Hispanic/Latino males.
police_killings %>% 
  ggplot()+
  geom_bar(aes(x=raceethnicity, fill=gender), position="dodge")


  1. Density plot for p_income (tract-level median personal income) and raceethnicity. This plot shows the median personal income by race. It appears to be that those who identify as Native American have the lowest median. Those who identify as White have the highest median personal income.
ggplot(data=police_killings) + 
  geom_density_ridges(aes(x=p_income, y=raceethnicity, fill=raceethnicity), alpha=0.25)


  1. Bar plot of armed, how/whether deceased was armed. Most people who were killed had firearm on them followed by those who were not armed at all.
ggplot(data=police_killings)+
  geom_bar(aes(x=armed), fill="lightblue")
